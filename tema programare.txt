#include <stdio.h>
int main(){
    int a , b , c ;
    printf("alege pretul produsului:");
    scanf("%d" , &a);
    printf("alege reducerea 1 2 sau 3 :");
    scanf("%d", &b);
    switch(b){
        case 1 :
          c = a*0.9 ;
          break;
        case 2 :
          c = a*0.8 ;
          break;
        case 3 :
          c = a*0.7 ;
          break;
        default :
          printf("nu ai ales bine reducerea\n");
          return 1;
          break;
        
    }
    printf("pretul final este : %d\n" , c);
    return 0;
}

    switch(cod){
        case 1 :
        impozit = 0.25;
        break;
        case 2 :
        impozit = 0.1;
        break;
        case 3 :
        impozit = 0.05;
        break;
        default:
        printf("nu e bun codul\n");
        return  1 ;
        break;
       
        
    }
