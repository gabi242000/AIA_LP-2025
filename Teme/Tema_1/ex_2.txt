import java.io.BufferedReader;
import java.io.*;
import java.math.*;

public static void main(String[] args) throws IOException{
    BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    System.out.println("Introdu 3 numere intregi");
    int a = Integer.parseInt(br.readLine());
    int b = Integer.parseInt(br.readLine());
    int c = Integer.parseInt(br.readLine());
    System.out.println("Suma numerelor " + (a+b+c));
    System.out.println("Media aritmetica " + (double)(a+b+c)/3);
    System.out.println("Numarul mare: " + Math.max(Math.max(a,b),c));
    System.out.println("Numarul mic: " + Math.min(Math.min(a,b),c));
}
