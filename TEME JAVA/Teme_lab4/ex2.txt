package temalab4ex2;
import java.util.*;
public class Masina {
    public String marca;
    public String model;
    public int an_fabricare;
    public String culoare;
    public int pret;
    public Masina(String marca, String model, int an_fabricare, String culoare, int pret) {
    	this.marca = marca;
    	this.model = model;
    	this.an_fabricare = an_fabricare;
    	this.culoare = culoare;
    	this.pret = pret;
    }
    public void afisareMasina() {
    	System.out.println("Marca: "+ marca);
    	System.out.println("Model: "+ model);
    	System.out.println("An fabricare: "+ an_fabricare);
    	System.out.println("Culoare: "+ culoare);
    	System.out.println("Pret: "+ pret +"lei");
    }
	public static void main(String[] args) {
	Scanner sc = new Scanner(System.in);
	System.out.print("Introdu marca masinii: ");
	String marca= sc.nextLine();
	System.out.print("Introdu modelul masinii: ");
	String model= sc.nextLine();
	System.out.print("Introdu anul de fabricare al masinii: ");
	int an_fabricare= sc.nextInt();
	sc.nextLine();
	System.out.print("Introdu culoarrea masinii: ");
	String culoare= sc.nextLine();
	System.out.print("Introdu pretul masinii: ");
	int pret= sc.nextInt();
	Masina m = new Masina(marca,model,an_fabricare,culoare,pret);
	System.out.println("Datele masinii sunt : ");
	m.afisareMasina();
	sc.close();
	}
}
