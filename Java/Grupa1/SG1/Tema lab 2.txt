import java.io.*;

public class Ex1 {
    public static void main(String[] args) throws IOException {
        int[] v = citireVector();
        afisarePare(v);
    }

    public static int[] citireVector() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        System.out.print("Introdu dimensiunea vectorului: ");
        int n = Integer.parseInt(br.readLine());
        int[] v = new int[n];

        for (int i = 0; i < n; i++) {
            System.out.print("Elementul " + (i + 1) + ": ");
            v[i] = Integer.parseInt(br.readLine());
        }
        return v;
    }

    public static void afisarePare(int[] v) {
        int suma = 0;
        System.out.print("Elementele pare sunt: ");
        for (int x : v) {
            if (x % 2 == 0) {
                System.out.print(x + " ");
                suma += x;
            }
        }
        System.out.println("\nSuma elementelor pare este: " + suma);
    }
}

import java.io.*;

public class Ex2 {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        System.out.print("Introdu un numar natural: ");
        int n = Integer.parseInt(br.readLine());

        int min = 9, max = 0;
        int copie = n;

        while (copie > 0) {
            int cifra = copie % 10;
            if (cifra < min) min = cifra;
            if (cifra > max) max = cifra;
            copie /= 10;
        }

        System.out.println("Cea mai mica cifra este: " + min);
        System.out.println("Cea mai mare cifra este: " + max);
    }
}
