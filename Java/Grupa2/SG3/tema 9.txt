import java.io.*;
import java.util.*;

public class Angajati {
    public static void main(String[] args) {
        File fisier = new File("angajati.txt");

        if (!fisier.exists()) {
            System.out.println("NU GASESC angajati.txt");
            return;
        }

        int nr = 0;
        double suma = 0;
        double maxSal = -1;
        String angajatMax = "";

        List<String> pesteMedie = new ArrayList<>();

        try (Scanner sc = new Scanner(fisier)) {
            while (sc.hasNext()) {
                String nume = sc.next();
                String prenume = sc.next();
                double salariu = sc.nextDouble();

                nr++;
                suma += salariu;

                if (salariu > maxSal) {
                    maxSal = salariu;
                    angajatMax = nume + " " + prenume;
                }

                pesteMedie.add(nume + " " + prenume + " " + salariu);
            }
        } catch (Exception e) {
            System.out.println("EROARE la citire!");
            return;
        }

        double medie = suma / nr;

        try (PrintWriter pw = new PrintWriter("venit_peste_medie.txt")) {
            for (String linie : pesteMedie) {
                String[] parts = linie.split(" ");
                double sal = Double.parseDouble(parts[2]);
                if (sal > medie)
                    pw.println(linie);
            }
        } catch (IOException e) {}

        try (PrintWriter pw = new PrintWriter("raport_angajati.txt")) {
            pw.println("Total angajati: " + nr);
            pw.println("Salariu mediu: " + medie);
            pw.println("Salariu maxim: " + maxSal + " (" + angajatMax + ")");
        } catch (IOException e) {}

        System.out.println("Angajati OK");
    }
}

import java.io.*;
import java.util.*;

public class Temperaturi {
    public static void main(String[] args) {
        File fisier = new File("temperaturi.txt");

        if (!fisier.exists()) {
            System.out.println("Fisierul temperaturi.txt nu exista!");
            return;
        }

        List<Double> temp = new ArrayList<>();
        double suma = 0;
        double min = Double.MAX_VALUE;
        double max = Double.MIN_VALUE;
        int subZero = 0;

        try (Scanner sc = new Scanner(fisier)) {
            while (sc.hasNextDouble()) {
                double t = sc.nextDouble();
                temp.add(t);
                suma += t;

                if (t < min) min = t;
                if (t > max) max = t;
                if (t < 0) subZero++;
            }
        } catch (Exception e) {
            System.out.println("Eroare la citire fisier!");
            return;
        }

        double medie = suma / temp.size();


        try (PrintWriter pw = new PrintWriter("raport_temperaturi.txt")) {
            pw.println("Temperatura medie: " + medie);
            pw.println("Temperatura minima: " + min);
            pw.println("Temperatura maxima: " + max);
            pw.println("Numar valori sub 0Â°C: " + subZero);
        } catch (IOException e) {
            System.out.println("Eroare la scriere raport_temperaturi.txt");
        }


        try (PrintWriter pw = new PrintWriter("temp_sub_zero.txt")) {
            for (double t : temp) {
                if (t < 0) {
                    pw.println(t);
                }
            }
        } catch (IOException e) {
            System.out.println("Eroare la scriere temp_sub_zero.txt");
        }

        System.out.println("Problema 2 rezolvata cu succes!");
    }
}
