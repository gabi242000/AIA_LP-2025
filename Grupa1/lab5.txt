#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main (){
    char s[101];
    fgets(s,sizeof(s),stdin);
    for (int i=0; s[i]!='\0';i++){
        s[i]=toupper(s[i]);
    }
    printf("%s", s);
    return 0;
}

    char t = s[x];

    printf("%s\n", s);
